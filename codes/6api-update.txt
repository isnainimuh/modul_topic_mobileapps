// Categories
static Future<List<Category>> getCategories() async {
  try {
    final response = await http.get(
      Uri.parse('$baseUrl/categories'),
      headers: await headers,
    );

    if (response.statusCode == 200) {
      final data = json.decode(response.body);
      if (data['success']) {
        return (data['data'] as List)
            .map((category) => Category.fromJson(category))
            .toList();
      }
    }
    throw Exception('Failed to load categories');
  } catch (e) {
    throw Exception('Network error: $e');
  }
}

static Future<Order> createOrder(String token, String shippingAddress) async {
  try {
    final response = await http.post(
      Uri.parse('$baseUrl/orders'),
      headers: await getHeaders(token: token),
      body: json.encode({
        'user_id': 1, // User ID sementara
        'shipping_address': shippingAddress,
      }),
    );

    if (response.statusCode == 200) {
      final data = json.decode(response.body);
      if (data['success']) {
        return Order.fromJson(data['data']['order']);
      }
    }
    throw Exception('Failed to create order');
  } catch (e) {
    throw Exception('Network error: $e');
  }
}

static Future<void> updatePaymentStatus(String token, int paymentId, String status) async {
  try {
    final response = await http.post(
      Uri.parse('$baseUrl/payments/$paymentId/status'),
      headers: await getHeaders(token: token),
      body: json.encode({
        'status': status,
      }),
    );

    if (response.statusCode != 200) {
      throw Exception('Failed to update payment status');
    }
  } catch (e) {
    throw Exception('Network error: $e');
  }
}
